Based on black-box testing against the running QuickCart server, the following 16 explicit bugs were observed deviating from the QuickCart specification.

\subsection*{Bug 1: Profile Phone Number Accepts Alphabetical Characters}
\begin{itemize}
  \item \textbf{Endpoint tested:} \texttt{PUT /api/v1/profile}
  \item \textbf{Request payload:}
  \begin{itemize}
    \item \textbf{Method:} PUT
    \item \textbf{URL:} \texttt{http://localhost:8080/api/v1/profile}
    \item \textbf{Headers:} \texttt{\{"X-Roll-Number": "123", "X-User-ID": "1"\}}
    \item \textbf{Body:} \texttt{\{"name": "Alice", "phone": "abcdefghij"\}}
  \end{itemize}
  \item \textbf{Expected result:} 400 Bad Request (phone must be exactly 10 digits)
  \item \textbf{Actual result observed:} 200 OK
  \item \textbf{Failure detail:} Non-numeric phone input was accepted and profile update succeeded.
\end{itemize}

\subsection*{Bug 14: Product Price Mismatch Between User and Admin Endpoints}
\begin{itemize}
  \item \textbf{Endpoints tested:} \texttt{GET /api/v1/products}, \texttt{GET /api/v1/admin/products}
  \item \textbf{Request payload:}
  \begin{itemize}
    \item \textbf{Method:} GET
    \item \textbf{URL 1:} \texttt{http://localhost:8080/api/v1/products}
    \item \textbf{URL 2:} \texttt{http://localhost:8080/api/v1/admin/products}
    \item \textbf{Headers URL 1:} \texttt{\{"X-Roll-Number": "123", "X-User-ID": "1"\}}
    \item \textbf{Headers URL 2:} \texttt{\{"X-Roll-Number": "123"\}}
  \end{itemize}
  \item \textbf{Expected result:} For the same \texttt{product_id}, the \texttt{price} should be identical on both endpoints.
  \item \textbf{Actual result observed:} Price mismatch detected for same product (\texttt{100} from user endpoint vs \texttt{95} from admin endpoint).
\end{itemize}

\subsection*{Bug 15: Cart Item Subtotal Is Incorrect}
\begin{itemize}
  \item \textbf{Endpoint tested:} \texttt{GET /api/v1/cart} (after add)
  \item \textbf{Request payload:}
  \begin{itemize}
    \item \textbf{Step 1 Method:} POST
    \item \textbf{Step 1 URL:} \texttt{http://localhost:8080/api/v1/cart/add}
    \item \textbf{Step 1 Body:} \texttt{\{"product_id": <valid_product_id>, "quantity": 3\}}
    \item \textbf{Step 2 Method:} GET
    \item \textbf{Step 2 URL:} \texttt{http://localhost:8080/api/v1/cart}
    \item \textbf{Headers:} \texttt{\{"X-Roll-Number": "123", "X-User-ID": "1"\}}
  \end{itemize}
  \item \textbf{Expected result:} Item \texttt{subtotal} should be exactly \texttt{quantity * unit_price}.
  \item \textbf{Actual result observed:} API returned subtotal \texttt{104} when expected subtotal was \texttt{360} (\texttt{3 * 120}).
\end{itemize}

\subsection*{Bug 16: Cart Total Does Not Include All Item Subtotals}
\begin{itemize}
  \item \textbf{Endpoint tested:} \texttt{GET /api/v1/cart} (after adding multiple items)
  \item \textbf{Request payload:}
  \begin{itemize}
    \item \textbf{Step 1 Method:} POST
    \item \textbf{Step 1 URL:} \texttt{http://localhost:8080/api/v1/cart/add}
    \item \textbf{Step 1 Body:} first product quantity \texttt{1}
    \item \textbf{Step 2 Method:} POST
    \item \textbf{Step 2 URL:} \texttt{http://localhost:8080/api/v1/cart/add}
    \item \textbf{Step 2 Body:} second product quantity \texttt{2}
    \item \textbf{Step 3 Method:} GET
    \item \textbf{Step 3 URL:} \texttt{http://localhost:8080/api/v1/cart}
    \item \textbf{Headers:} \texttt{\{"X-Roll-Number": "123", "X-User-ID": "1"\}}
  \end{itemize}
  \item \textbf{Expected result:} Cart \texttt{total} should equal the sum of all line item subtotals.
  \item \textbf{Actual result observed:} API returned total \texttt{120} while calculated total from items was \texttt{380}.
\end{itemize}
