\section*{Errors Found and Fixes}

\subsection*{Error \#1: GO salary not awarded when passing start}
\begin{itemize}
  \item \textbf{Observed:} Moving from position 39 to 1 did not increase balance.
  \item \textbf{Root cause:} Salary granted only when \texttt{position == 0}.
  \item \textbf{Fix:} Use old/new position wrap-around detection for positive moves.
  \item \textbf{Commit:} \texttt{a545f03} - \texttt{Error \#1: fix GO salary when passing start}
\end{itemize}

\subsection*{Error \#2: Full-group ownership check accepted partial ownership}
\begin{itemize}
  \item \textbf{Observed:} Group reported owned when only one property matched owner.
  \item \textbf{Root cause:} Used \texttt{any(...)} instead of all-properties check.
  \item \textbf{Fix:} Require non-empty set and \texttt{all(p.owner == player ...)}.
  \item \textbf{Commit:} \texttt{a78021a} - \texttt{Error \#2: correct full-set ownership logic for property groups}
\end{itemize}

\subsection*{Error \#3: Exact-balance purchase rejected}
\begin{itemize}
  \item \textbf{Observed:} Player with balance equal to price could not buy.
  \item \textbf{Root cause:} Affordability check used \texttt{<=} reject branch.
  \item \textbf{Fix:} Change condition to \texttt{<} so equality is valid.
  \item \textbf{Commit:} \texttt{2cc419b} - \texttt{Error \#3: allow buying property with exact available balance}
\end{itemize}

\subsection*{Error \#4: Rent not credited to owner}
\begin{itemize}
  \item \textbf{Observed:} Tenant lost money but owner balance remained unchanged.
  \item \textbf{Root cause:} Missing owner credit operation in rent flow.
  \item \textbf{Fix:} Add \texttt{prop.owner.add\_money(rent)}.
  \item \textbf{Commit:} \texttt{5c09312} - \texttt{Error \#4: transfer rent to owner when tenant pays}
\end{itemize}

\subsection*{Error \#5: Winner selected by lowest wealth}
\begin{itemize}
  \item \textbf{Observed:} Winner function returned minimum net worth player.
  \item \textbf{Root cause:} Used \texttt{min(...)} for winner computation.
  \item \textbf{Fix:} Replace with \texttt{max(...)} by net worth.
  \item \textbf{Commit:} \texttt{d6a35d7} - \texttt{Error \#5: choose winner by highest net worth}
\end{itemize}
