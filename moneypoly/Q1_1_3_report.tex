\section*{Errors Found and Fixes (Required Commit Format)}

\subsection*{Error \#1: GO salary not awarded when passing start}
\begin{itemize}
  \item Root cause: salary branch only checked \texttt{position == 0}
  \item Commit: \texttt{a545f03} -- \texttt{Error \#1: fix GO salary when passing start}
\end{itemize}

\subsection*{Error \#2: Property-group ownership accepted partial ownership}
\begin{itemize}
  \item Root cause: used \texttt{any(...)} instead of all-properties check
  \item Commit: \texttt{a78021a} -- \texttt{Error \#2: correct full-set ownership logic for property groups}
\end{itemize}

\subsection*{Error \#3: Exact-balance purchase rejected}
\begin{itemize}
  \item Root cause: affordability check used \texttt{<=}
  \item Commit: \texttt{2cc419b} -- \texttt{Error \#3: allow buying property with exact available balance}
\end{itemize}

\subsection*{Error \#4: Rent not credited to owner}
\begin{itemize}
  \item Root cause: missing owner credit operation in \texttt{pay\_rent}
  \item Commit: \texttt{5c09312} -- \texttt{Error \#4: transfer rent to owner when tenant pays}
\end{itemize}

\subsection*{Error \#5: Winner selected by lowest net worth}
\begin{itemize}
  \item Root cause: used \texttt{min(...)} instead of \texttt{max(...)}
  \item Commit: \texttt{d6a35d7} -- \texttt{Error \#5: choose winner by highest net worth}
\end{itemize}

\subsection*{Error \#6: Empty card deck crashed in cards\_remaining}
\begin{itemize}
  \item Root cause: modulo-by-zero when deck length was zero
  \item Commit: \texttt{15766ab} -- \texttt{Error \#6: handle empty deck in cards\_remaining}
\end{itemize}

\subsection*{Error \#7: Unmortgage could clear state before affordability validation}
\begin{itemize}
  \item Root cause: \texttt{unmortgage()} was called before balance check
  \item Commit: \texttt{5ccaa98} -- \texttt{Error \#7: prevent unmortgage state loss on insufficient funds}
\end{itemize}
